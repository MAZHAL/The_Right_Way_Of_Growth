## json 

- struct  <=>   json
```golang 
  
  type Person struct {
    
   //"-"是忽略的意思
        Name  string `json:"-"`
        Hobby string `json:"hobby" `
    
  }
  
  
   p := Person{"5lmh.com", "女"}
    // 编码json
    b, err := json.Marshal(p)
    if err != nil {
        fmt.Println("json err ", err)
    }
    fmt.Println(string(b))

    // 格式化输出
    b, err = json.MarshalIndent(p, "", "     ")
    if err != nil {
        fmt.Println("json err ", err)
    }
    fmt.Println(string(b))   // json 格式数据
    
    var p2 Person 
    json.Unmarshal(b,&p2)
    fmt.Println(p2)   // person 结构体

```
